Create database  BD_T2_APELLIDO;
 
 use BD_T2_APELLIDO;
 
 create table categoria (
 id_cate int primary key ,
 descripcion varchar(100) not null,
 frecuencia_compra int
 );
 
 create table producto(
 id_prod int primary key auto_increment,
 nom_prod varchar(100) not null,
 id_cate int,
 stock_actual int not null,
 foreign key (id_cate) references categoria(id_cate)
 );
 
create table inventario(
nro_inventario int primary key auto_increment,
fecha date not null,
id_prod int,
costo_ingreso decimal(10,2) not null,
foreign key (id_prod) references producto(id_prod)
);
